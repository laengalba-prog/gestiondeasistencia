  <script>
    function mostrarSeccion(id) {
      document.querySelectorAll('main').forEach(sec => sec.classList.add('oculto'));
      document.getElementById(id).classList.remove('oculto');
    }
    function cambiarVista() {
      const vista = document.getElementById('vista').value;
      document.getElementById('vistaSemanal').classList.toggle('oculto', vista !== 'semanal');
      document.getElementById('vistaMensual').classList.toggle('oculto', vista !== 'mensual');
    }
